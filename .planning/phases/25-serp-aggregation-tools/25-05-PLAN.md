2. **POST /{slug}/rerun/{job_id}** — Re-run endpoint:
   ```python
   @router.post("/{slug}/rerun/{job_id}", name="tool_rerun")
   async def tool_rerun(slug, job_id, db, user):
       JobModel, _ = _get_tool_models(slug)
       old_job = await db.get(JobModel, uuid.UUID(str(job_id)))
       if not old_job:
           raise HTTPException(404)
       # Create new job with same input
       new_kwargs = {
           "id": uuid.uuid4(),
           "status": "pending",
           "user_id": user.id,
           "created_at": datetime.now(timezone.utc),
       }
       # Copy input fields based on registry
       registry = TOOL_REGISTRY[slug]
       new_kwargs[registry["input_col"]] = getattr(old_job, registry["input_col"])
       new_kwargs[registry["count_col"]] = getattr(old_job, registry["count_col"])
       if registry.get("has_domain_field"):
           new_kwargs["target_domain"] = getattr(old_job, "target_domain", "")
       if registry.get("has_region_field"):
           new_kwargs["input_region"] = getattr(old_job, "input_region", 213)
       new_job = JobModel(**new_kwargs)
       db.add(new_job)
       await db.commit()
       # Dispatch task (same logic as tool_submit)
       if slug == "brief":
           from celery import chain as celery_chain
           from app.tasks.brief_tasks import (run_brief_step1_serp, run_brief_step2_crawl, run_brief_step3_aggregate, run_brief_step4_finalize)
           celery_chain(run_brief_step1_serp.si(str(new_job.id)), run_brief_step2_crawl.si(str(new_job.id)), run_brief_step3_aggregate.si(str(new_job.id)), run_brief_step4_finalize.si(str(new_job.id))).delay()
       else:
           task_fn = _get_tool_task(slug)
           task_fn.delay(str(new_job.id))
       return RedirectResponse(f"/ui/tools/{slug}/{new_job.id}", status_code=303)
   ```
   Add `HX-Redirect` header for HTMX: `response.headers["HX-Redirect"] = f"/ui/tools/{slug}/{new_job.id}"`
