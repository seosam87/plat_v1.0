**2. Create `app/services/mobile_brief_service.py`** (new file):

```python
from __future__ import annotations
import uuid
from jinja2 import Environment, FileSystemLoader
from loguru import logger
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession
from app.config import settings
from app.models.client import Client

_env = Environment(loader=FileSystemLoader("app/templates"))


def render_brief(
    project_name: str,
    site_url: str,
    length: str,
    tone: str,
    keywords: list[str],
) -> str:
    """Render copywriter brief to plain text string."""
    tmpl = _env.get_template("briefs/copywriter_brief.txt.j2")
    return tmpl.render(
        project_name=project_name,
        site_url=site_url,
        length=length,
        tone=tone,
        keywords=keywords,
    )


async def list_clients_for_brief(db: AsyncSession) -> list[dict]:
    """List clients with email for recipient select (per D-11).
    Uses Client.email IS NOT NULL only (Pitfall 4 from RESEARCH — Client has no telegram_username)."""
    stmt = select(Client).where(Client.email.isnot(None)).order_by(Client.name)
    result = await db.execute(stmt)
    clients = result.scalars().all()
    return [{"id": str(c.id), "name": c.name, "email": c.email} for c in clients]


async def send_brief_telegram(text: str, client_email: str) -> bool:
    """Send brief text via Telegram. For Phase 30 MVP, text briefs <4000 chars
    are sent as inline messages. Longer briefs sent as .txt attachment.
    Uses existing notification service pattern."""
    # Phase 30 MVP: log intent, actual Telegram send uses bot API if configured
    logger.info(f"Brief Telegram send requested for {client_email}, text length={len(text)}")
    # Reuse pattern from mobile_reports_service Telegram delivery
    # If telegram bot not configured, return False gracefully
    try:
        if not settings.TELEGRAM_BOT_TOKEN:
            logger.warning("Telegram bot not configured, skipping brief send")
            return False
        # Import here to avoid circular dependency
        from app.services.telegram_service import send_message_to_user_by_email
        if len(text) <= 4000:
            await send_message_to_user_by_email(client_email, text)
        else:
            # For long texts, truncate with link note
            truncated = text[:3900] + "\n\n[Полный текст ТЗ доступен в платформе]"
            await send_message_to_user_by_email(client_email, truncated)
        return True
    except Exception as e:
        logger.error(f"Telegram brief send failed: {e}")
        return False


async def send_brief_email(text: str, client_email: str, project_name: str) -> bool:
    """Send brief text via email. Plain text body, no attachment for <4000 chars."""
    logger.info(f"Brief email send requested for {client_email}, text length={len(text)}")
    try:
        # Reuse email sending pattern from existing codebase
        import aiosmtplib
        from email.message import EmailMessage
        msg = EmailMessage()
        msg["Subject"] = f"ТЗ копирайтеру: {project_name}"
        msg["From"] = settings.SMTP_FROM or "noreply@seo-platform.local"
        msg["To"] = client_email
        if len(text) <= 4000:
            msg.set_content(text)
        else:
            msg.set_content("ТЗ копирайтеру прикреплено к письму.")
            msg.add_attachment(text.encode("utf-8"), maintype="text", subtype="plain", filename="brief.txt")
        await aiosmtplib.send(
            msg,
            hostname=settings.SMTP_HOST or "localhost",
            port=int(settings.SMTP_PORT or 587),
            use_tls=bool(settings.SMTP_USE_TLS),
        )
        return True
    except Exception as e:
        logger.error(f"Email brief send failed: {e}")
        return False
```

Note: The Telegram and email send functions have graceful error handling — if services not configured, they return False without crashing. The router will show appropriate toast based on return value.

**mode=brief flow (per D-11):**
1. Load Project. If `Project.site_id is None` → same 422 toast.

   - `site_id = project.site_id`
   - `task_type = TaskType.manual`
   - `priority = TaskPriority(priority)`
   - `project_id = UUID(project_id)`
6. `db.add(task)`, `await db.commit()`.
7. If recipient_id and action == "send":
   - Load Client by recipient_id.
   - Try Telegram: `await send_brief_telegram(rendered, client.email)`.
   - Try Email: `await send_brief_email(rendered, client.email, project.name)`.
   - Toast: "ТЗ создано и отправлено клиенту" if any send succeeded, else "ТЗ сохранено (отправка не удалась)".
8. Else toast: "ТЗ сохранено".
9. Return HTML with showToast + redirect `/m/`.
