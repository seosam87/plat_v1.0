```python
@router.get("/api/playbook-step-active")
async def api_playbook_step_active(
    step_id: uuid.UUID,
    user: User = Depends(require_any_authenticated),
    db: AsyncSession = Depends(get_async_session),
):
    """Fetch step data for banner rendering."""
    step = await svc.get_project_step(db, step_id)
    if step is None:
        raise HTTPException(404, "Step not found")
    pp = step.project_playbook
    total = len(pp.steps) if pp and pp.steps else 0
    # Count the position of this step within its playbook (for "Шаг N/M" display)
    return {
        "step_id": str(step.id),
        "project_playbook_id": str(step.project_playbook_id),
        "project_id": str(pp.project_id) if pp else None,
        "title": step.block.title,
        "position": step.position + 1,
        "total": total,
        "playbook_name": pp.name if pp else "",
        "status": step.status.value,
        "action_kind": step.block.action_kind.value,
    }


@router.post("/api/project-playbook-steps/{step_id}/complete")
```

```javascript
// Phase 999.8 Playbook Banner (D-16/D-17).
// Reads sessionStorage.active_playbook_step on DOMContentLoaded, fetches step data,
// renders the sticky banner, and handles the two CTA buttons.
(function () {
  'use strict';

  const SS_KEY = 'active_playbook_step';

  function hide() {
    const el = document.getElementById('playbook-banner');
    if (el) { el.hidden = true; el.style.display = 'none'; }
    document.body.classList.remove('has-playbook-banner');
  }

  function show(title, meta) {
    const el = document.getElementById('playbook-banner');
    if (!el) return;
    document.getElementById('playbook-banner-title').textContent = title;
    document.getElementById('playbook-banner-meta').textContent = meta;
    el.hidden = false;
    el.style.display = 'flex';
    document.body.classList.add('has-playbook-banner');
  }

  async function init() {
    // Skip on mobile routes entirely
    if (window.location.pathname.startsWith('/m/')) {
      hide();
      return;
    }
    let state;
    try {
      const raw = sessionStorage.getItem(SS_KEY);
      if (!raw) { hide(); return; }
      state = JSON.parse(raw);
    } catch (e) {
      console.warn('playbook_banner: bad sessionStorage', e);
      sessionStorage.removeItem(SS_KEY);
      hide();
      return;
    }
    if (!state || !state.step_id) { hide(); return; }

    // Fetch fresh step data
    try {
      const resp = await fetch('/api/playbook-step-active?step_id=' + encodeURIComponent(state.step_id), {
        headers: {'Accept': 'application/json'},
      });
      if (resp.status === 404) {
        // Step gone (completed elsewhere) — silently clear
        sessionStorage.removeItem(SS_KEY);
        hide();
        return;
      }
      if (!resp.ok) { hide(); return; }
      const data = await resp.json();
      if (data.status === 'done') {
        sessionStorage.removeItem(SS_KEY);
        hide();
        return;
      }
      const title = 'Шаг ' + data.position + '/' + data.total + ' · ' + data.title;
      const meta = 'Playbook: ' + data.playbook_name;
      show(title, meta);
      wireButtons(data);
    } catch (e) {
      console.warn('playbook_banner: fetch failed', e);
      hide();
    }
  }

  function wireButtons(data) {
    const complete = document.getElementById('playbook-banner-complete');
    const back = document.getElementById('playbook-banner-back');
    if (complete) {
      complete.onclick = async function () {
        try {
          const r = await fetch('/api/project-playbook-steps/' + data.step_id + '/complete', {method: 'POST'});
          if (r.ok) {
            sessionStorage.removeItem(SS_KEY);
            const body = await r.json().catch(() => ({}));
            const projectId = body.project_id || data.project_id;
            if (projectId) {
              window.location.href = '/ui/projects/' + projectId + '/kanban#playbook';
            } else {
              hide();
            }
          } else {
            alert('Не удалось отметить выполненным. Попробуйте ещё раз.');
          }
        } catch (e) {
          alert('Ошибка сети. Попробуйте ещё раз.');
        }
      };
    }
    if (back) {
      back.onclick = function () {
        sessionStorage.removeItem(SS_KEY);
        if (data.project_id) {
          window.location.href = '/ui/projects/' + data.project_id + '/kanban#playbook';
        } else {
          hide();
        }
      };
    }
  }

  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', init);
  } else {
    init();
  }
})();
```
