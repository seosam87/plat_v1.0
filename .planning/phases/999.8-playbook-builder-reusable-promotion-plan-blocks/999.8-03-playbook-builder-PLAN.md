async def list_playbooks(
    db: AsyncSession,
    *,
    category: PlaybookCategory | None = None,
    only_published: bool = False,
) -> Sequence[Playbook]:
    stmt = select(Playbook).options(
        selectinload(Playbook.steps).selectinload(PlaybookStep.block)
    ).order_by(Playbook.name)
    if category is not None:
        stmt = stmt.where(Playbook.category == category)
    if only_published:
        stmt = stmt.where(Playbook.is_published.is_(True))
    result = await db.execute(stmt)
    return result.scalars().unique().all()


async def get_playbook_with_steps(db: AsyncSession, playbook_id: uuid.UUID) -> Playbook | None:
    stmt = (
        select(Playbook)
        .options(
            selectinload(Playbook.steps).selectinload(PlaybookStep.block).selectinload(PlaybookBlock.category),
            selectinload(Playbook.steps).selectinload(PlaybookStep.block).selectinload(PlaybookBlock.expert_source),
        )
        .where(Playbook.id == playbook_id)
    )
    result = await db.execute(stmt)
    return result.scalar_one_or_none()


async def create_playbook(db: AsyncSession, data: PlaybookCreate, created_by: uuid.UUID | None) -> Playbook:
    slug = await _unique_slug(db, Playbook, _slugify(data.name))
    playbook = Playbook(
        name=data.name,
        slug=slug,
        description_md=data.description_md,
        category=data.category,
        is_published=data.is_published,
        created_by=created_by,
    )
    db.add(playbook)
    await db.commit()
    await db.refresh(playbook)
    return playbook


async def update_playbook(db: AsyncSession, playbook_id: uuid.UUID, data: PlaybookUpdate) -> Playbook | None:
    pb = await get_playbook_with_steps(db, playbook_id)
    if pb is None:
        return None
    pb.name = data.name
    pb.description_md = data.description_md
    pb.category = data.category
    pb.is_published = data.is_published
    await db.commit()
    await db.refresh(pb)
    return pb


async def delete_playbook(db: AsyncSession, playbook_id: uuid.UUID) -> bool:
    pb = await get_playbook_with_steps(db, playbook_id)
    if pb is None:
        return False
    await db.delete(pb)  # cascade drops steps
    await db.commit()
    return True


async def add_step(
    db: AsyncSession,
    playbook_id: uuid.UUID,
    block_id: uuid.UUID,
) -> PlaybookStep | None:
    pb = await get_playbook_with_steps(db, playbook_id)
    if pb is None:
        return None
    # next position
    max_pos = max((s.position for s in pb.steps), default=-1)
    step = PlaybookStep(
        playbook_id=playbook_id,
        block_id=block_id,
        position=max_pos + 1,
    )
    db.add(step)
    await db.commit()
    await db.refresh(step)
    # eager-load block for rendering
    result = await db.execute(
        select(PlaybookStep)
        .options(
            selectinload(PlaybookStep.block).selectinload(PlaybookBlock.category),
            selectinload(PlaybookStep.block).selectinload(PlaybookBlock.expert_source),
        )
        .where(PlaybookStep.id == step.id)
    )
    return result.scalar_one()


async def remove_step(db: AsyncSession, step_id: uuid.UUID) -> bool:
    result = await db.execute(select(PlaybookStep).where(PlaybookStep.id == step_id))
    step = result.scalar_one_or_none()
    if step is None:
        return False
    playbook_id = step.playbook_id
    await db.delete(step)
    await db.flush()
    # Re-compact positions for remaining steps
    remaining = await db.execute(
        select(PlaybookStep).where(PlaybookStep.playbook_id == playbook_id).order_by(PlaybookStep.position)
    )
    for i, s in enumerate(remaining.scalars().all()):
        s.position = i
    await db.commit()
    return True


async def reorder_steps(
    db: AsyncSession,
    playbook_id: uuid.UUID,
    step_ids_in_order: list[uuid.UUID],
) -> bool:
    """Apply a new ordering. Uses two-phase update to avoid unique constraint clash."""
    result = await db.execute(
        select(PlaybookStep).where(PlaybookStep.playbook_id == playbook_id)
    )
    steps = {s.id: s for s in result.scalars().all()}
    if set(steps.keys()) != set(step_ids_in_order):
        return False
    # Phase 1: offset all positions by +1000 to escape the unique constraint
    for s in steps.values():
        s.position = s.position + 1000
    await db.flush()
    # Phase 2: assign final positions
    for i, sid in enumerate(step_ids_in_order):
        steps[sid].position = i
    await db.commit()
    return True


async def move_step(db: AsyncSession, playbook_id: uuid.UUID, step_id: uuid.UUID, direction: str) -> bool:
    """Keyboard-accessible fallback — move a step up or down by one."""
    result = await db.execute(
        select(PlaybookStep).where(PlaybookStep.playbook_id == playbook_id).order_by(PlaybookStep.position)
    )
    steps = list(result.scalars().all())
    ids = [s.id for s in steps]
    if step_id not in ids:
        return False
    idx = ids.index(step_id)
    if direction == "up" and idx > 0:
        ids[idx - 1], ids[idx] = ids[idx], ids[idx - 1]
    elif direction == "down" and idx < len(ids) - 1:
        ids[idx], ids[idx + 1] = ids[idx + 1], ids[idx]
    else:
        return True  # no-op
    return await reorder_steps(db, playbook_id, ids)


async def update_step_note(db: AsyncSession, step_id: uuid.UUID, note_md: str | None) -> bool:
    result = await db.execute(select(PlaybookStep).where(PlaybookStep.id == step_id))
    step = result.scalar_one_or_none()
    if step is None:
        return False
    step.note_md = note_md or None
    await db.commit()
    return True


async def clone_playbook(db: AsyncSession, playbook_id: uuid.UUID, created_by: uuid.UUID | None) -> Playbook | None:
    src = await get_playbook_with_steps(db, playbook_id)
    if src is None:
        return None
    new_slug = await _unique_slug(db, Playbook, f"{src.slug}-copy")
    new_pb = Playbook(
        name=f"{src.name} (копия)",
        slug=new_slug,
        description_md=src.description_md,
        category=src.category,
        is_published=False,  # clones start unpublished for safety
        created_by=created_by,
    )
    db.add(new_pb)
    await db.flush()
    for s in src.steps:
        db.add(PlaybookStep(
            playbook_id=new_pb.id,
            block_id=s.block_id,
            position=s.position,
            note_md=s.note_md,
        ))
    await db.commit()
    await db.refresh(new_pb)
    return new_pb

<script>
(function() {
  const list = document.getElementById('steps-list');
  if (!list) return;
  const playbookId = list.dataset.playbookId;
  new Sortable(list, {
    handle: '.drag-handle',
    animation: 150,
    onEnd: function() {
      const order = Array.from(list.querySelectorAll('[data-step-id]')).map(el => el.dataset.stepId);
      fetch(`/ui/playbooks/${playbookId}/steps/reorder`, {
        method: 'POST',
        headers: {'Content-Type': 'application/json'},
        body: JSON.stringify({order: order}),
      }).then(r => {
        if (r.ok) {
          window.dispatchEvent(new CustomEvent('toast', {detail: {msg: 'Порядок сохранён', type: 'success'}}));
        } else {
          window.dispatchEvent(new CustomEvent('toast', {detail: {msg: 'Ошибка сохранения порядка', type: 'error'}}));
          setTimeout(() => location.reload(), 1500);
        }
      });
    }
  });
})();
</script>
