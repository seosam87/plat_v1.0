```python
def _slugify(text: str) -> str:
    """Transliterate Cyrillic and slugify; never empty."""
    # Minimal cyrillic → latin table (sufficient for MVP)
    table = str.maketrans({
        "а":"a","б":"b","в":"v","г":"g","д":"d","е":"e","ё":"e","ж":"zh","з":"z",
        "и":"i","й":"i","к":"k","л":"l","м":"m","н":"n","о":"o","п":"p","р":"r",
        "с":"s","т":"t","у":"u","ф":"f","х":"kh","ц":"ts","ч":"ch","ш":"sh","щ":"sch",
        "ъ":"","ы":"y","ь":"","э":"e","ю":"yu","я":"ya",
    })
    s = text.lower().translate(table)
    s = re.sub(r"[^a-z0-9]+", "-", s).strip("-")
    return s or uuid.uuid4().hex[:8]

async def _unique_slug(db: AsyncSession, model, base: str) -> str:
    """Append -2, -3, ... until unique."""
    slug = base
    n = 1
    while True:
        result = await db.execute(select(model).where(model.slug == slug))
        if result.scalar_one_or_none() is None:
            return slug
        n += 1
        slug = f"{base}-{n}"
```

```python
        bio_md=data.bio_md,
        external_url=str(data.external_url) if data.external_url else None,
    )
    db.add(expert)
    await db.commit()
    await db.refresh(expert)
    return expert

async def update_expert(db: AsyncSession, expert_id: uuid.UUID, data: ExpertUpdate) -> ExpertSource | None:

    if expert is None:
        return None
    expert.name = data.name
    expert.bio_md = data.bio_md
    expert.external_url = str(data.external_url) if data.external_url else None
    await db.commit()
    await db.refresh(expert)
    return expert

async def delete_expert(db: AsyncSession, expert_id: uuid.UUID) -> bool:

async def list_blocks(
    db: AsyncSession,
    *,
    category_id: uuid.UUID | None = None,
    expert_source_id: uuid.UUID | None = None,
) -> Sequence[PlaybookBlock]:
    stmt = (
        select(PlaybookBlock)
        .options(
            selectinload(PlaybookBlock.category),
            selectinload(PlaybookBlock.expert_source),
            selectinload(PlaybookBlock.media),
        )
        .order_by(PlaybookBlock.display_order, PlaybookBlock.title)
    )
    if category_id is not None:
        stmt = stmt.where(PlaybookBlock.category_id == category_id)
    if expert_source_id is not None:
        stmt = stmt.where(PlaybookBlock.expert_source_id == expert_source_id)
    result = await db.execute(stmt)
    return result.scalars().unique().all()

async def get_block(db: AsyncSession, block_id: uuid.UUID) -> PlaybookBlock | None:
    result = await db.execute(
        select(PlaybookBlock)
        .options(
            selectinload(PlaybookBlock.category),
            selectinload(PlaybookBlock.expert_source),
            selectinload(PlaybookBlock.media),
        )
        .where(PlaybookBlock.id == block_id)
    )
    return result.scalar_one_or_none()

async def create_block(
    db: AsyncSession,
    data: BlockCreate,
    media: list[BlockMediaCreate],
    created_by: uuid.UUID | None,
) -> PlaybookBlock:
    slug = await _unique_slug(db, PlaybookBlock, _slugify(data.title))
    block = PlaybookBlock(
        title=data.title,
        slug=slug,
        category_id=data.category_id,

        estimated_days=data.estimated_days,
        display_order=data.display_order,
        prerequisites=[],
        created_by=created_by,

    await db.flush()
    for idx, m in enumerate(media):
        db.add(BlockMedia(
            block_id=block.id,
            kind=m.kind,
            url=str(m.url),
            title=m.title,
            description_md=m.description_md,
            display_order=idx,

    await db.commit()
    await db.refresh(block)
    return block
```

```python
def _parse_media_rows(form) -> list[BlockMediaCreate]:
    """Parse repeated form fields media[i][kind], media[i][url], etc."""
    out: list[BlockMediaCreate] = []
    # Form field names: media_kind, media_url, media_title, media_description (list-style via getlist)
    kinds = form.getlist("media_kind") if hasattr(form, "getlist") else []
    urls = form.getlist("media_url") if hasattr(form, "getlist") else []
    titles = form.getlist("media_title") if hasattr(form, "getlist") else []
    descs = form.getlist("media_description") if hasattr(form, "getlist") else []
    for i in range(len(urls)):
        if not urls[i].strip():
            continue
        out.append(BlockMediaCreate(
            kind=BlockMediaKind(kinds[i]) if i < len(kinds) and kinds[i] else BlockMediaKind.article,
            url=urls[i].strip(),
            title=(titles[i] if i < len(titles) else "") or urls[i],
            description_md=(descs[i] if i < len(descs) else None) or None,
        ))
    return out

    experts = await svc.list_experts(db)
    counts = {e.id: await svc.count_blocks_by_expert(db, e.id) for e in experts}

@router.post("/ui/playbooks/experts/new")
```
