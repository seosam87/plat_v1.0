    ```python
    """Smoke crawler helpers — route discovery, param resolution, response assertions.
    Per Phase 15.1 CONTEXT D-02 / D-03 / D-04.
    """
    from __future__ import annotations
    import re
    from dataclasses import dataclass
    from fastapi import FastAPI
    from fastapi.routing import APIRoute

    JINJA_ERROR_MARKERS = (
        "TemplateSyntaxError",
        "UndefinedError",
        "jinja2.exceptions",
        "Traceback (most recent call last)",
        "builtin_function_or_method",
    )

    SMOKE_SKIP: dict[str, str] = {
        "/ui/login": "auth entry point — redirect loop in tests",
        "/ui/logout": "session destruction",
        "/ui/logs": "live tail endpoint",
        # wp_pipeline routes use {job_id} which collides with SuggestJob — defer to scenario runner (backlog 999.1)
        # Add specific paths once enumerated; helper test will fail if you forget any.
    }

    UI_PREFIXES = ("/ui/", "/analytics/")

    class UnknownParamError(RuntimeError):
        pass

    @dataclass(frozen=True)
    class RouteSpec:
        path: str
        name: str
        methods: frozenset[str]

    def discover_routes(app: FastAPI) -> list[RouteSpec]:
        out: list[RouteSpec] = []
        for r in app.routes:
            if not isinstance(r, APIRoute):
                continue
            if "GET" not in r.methods:
                continue
            if not any(r.path.startswith(p) for p in UI_PREFIXES):
                continue
            if r.path in SMOKE_SKIP:
                continue
            # Skip wp_pipeline routes via prefix match (job_id collision)
            if r.path.startswith("/ui/wp-pipeline/") and "{job_id}" in r.path:
                continue
            out.append(RouteSpec(path=r.path, name=r.name or r.path, methods=frozenset(r.methods)))
        return sorted(out, key=lambda s: s.path)

    def build_param_map(smoke_ids: dict[str, str]) -> dict[str, str]:
        pm = dict(smoke_ids)
        pm["module"] = "sites"  # /ui/help/{module} — string param, NOT uuid (per RESEARCH correction)
        return pm

    _PARAM_RE = re.compile(r"\{([^}:]+)(?::[^}]+)?\}")

    def resolve_path(path_template: str, param_map: dict[str, str]) -> str:
        def repl(m: re.Match) -> str:
            name = m.group(1)
            if name not in param_map:
                raise UnknownParamError(
                    f"Unknown path param '{name}' in {path_template} — add to PARAM_MAP or SMOKE_SKIP"
                )
            return param_map[name]
        return _PARAM_RE.sub(repl, path_template)

    def assert_no_error_markers(body: str, path: str) -> None:
        for marker in JINJA_ERROR_MARKERS:
            if marker in body:
                raise AssertionError(f"{path}: response body contains error marker '{marker}'")

    def assert_structural_html(body: str, path: str, *, is_partial: bool) -> None:
        if is_partial:
            if not body or body.strip() == "":
                raise AssertionError(f"{path}: partial returned empty body")
            return
        if "<title>" not in body:
            raise AssertionError(f"{path}: full page missing <title>")
        if "<body" not in body and "<main" not in body:
            raise AssertionError(f"{path}: full page missing <body>/<main>")

    def is_partial(path: str) -> bool:
        return "/tabs/" in path or "/partials/" in path or "/detail/" in path
    ```
