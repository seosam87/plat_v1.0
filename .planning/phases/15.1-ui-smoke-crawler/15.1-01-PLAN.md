    Critical structural pattern (per RESEARCH "Critical detail"):

    ```python
    @pytest_asyncio.fixture(scope="session")
    async def smoke_seed():
        engine = create_async_engine(TEST_DATABASE_URL, echo=False)
        # Create all tables (idempotent)
        async with engine.begin() as conn:
            await conn.run_sync(Base.metadata.create_all)

        # OUTER connection + OUTER transaction held for whole session
        conn = await engine.connect()
        outer_trans = await conn.begin()

        # Session joins outer transaction; downstream consumers will use
        # join_transaction_mode="create_savepoint" so per-request commits
        # become savepoints rolled back at request end.
        Session = async_sessionmaker(
            bind=conn,
            expire_on_commit=False,
            join_transaction_mode="create_savepoint",
        )
        session = Session()

        try:
            # --- Insert CORE rows via ORM ---
            # Site, admin User (bcrypt hashed), KeywordGroup, 10 Keywords,
            # 5 KeywordLatestPosition, 2 GapKeyword, 1 SuggestJob (status complete),
            # 1 CrawlJob (status done), 1 ClientReport (status ready),
            # 4 ServiceCredential (wordstat, gsc, xmlproxy, rucaptcha),
            # 1 AuditCheckDefinition(id=UUID(SMOKE_IDS["audit_check_id"]), code="smoke_check", ...),
            # 1 AuditResult(id=UUID(SMOKE_IDS["audit_id"]), check_definition_id=UUID(SMOKE_IDS["audit_check_id"]), site_id=UUID(SMOKE_IDS["site_id"]), ...)
            await session.flush()
            yield SeedHandle(ids=SMOKE_IDS, session=session, connection=conn)
        finally:
            await session.close()
            await outer_trans.rollback()
            await conn.close()
            await engine.dispose()
    ```
