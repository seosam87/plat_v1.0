**API client functions (async, httpx):**

1. `async def fetch_daily_traffic(counter_id: str, token: str, date1: str, date2: str) -> list[dict]`
   - Calls `GET {API_BASE}/data/bytime` with `group=day`, metrics = `ym:s:visits,ym:s:bounceRate,ym:s:pageDepth,ym:s:avgVisitDurationSeconds`, filter = ORGANIC_FILTER
   - Parses response: extract `time_intervals` and `data[0].metrics` arrays
   - Returns list of dicts: `{"date": "YYYY-MM-DD", "visits": int, "bounce_rate": float, "page_depth": float, "avg_duration_seconds": int}`
   - Also adds `ym:s:users` to metrics and returns `"users"` key
   - Raises `httpx.HTTPStatusError` on failure (caller handles)

2. `async def fetch_page_traffic(counter_id: str, token: str, date1: str, date2: str, limit: int = 500) -> list[dict]`
   - Calls `GET {API_BASE}/data` with dimension `ym:s:startURL`, metrics, filter = ORGANIC_FILTER, sort = `-ym:s:visits`
   - Pagination loop: increment offset by limit until `len(all_rows) >= total_rows`
   - URL normalization: strip query parameters from `page_url` (split on `?`, take first part)
   - Returns list of dicts: `{"page_url": str, "visits": int, "bounce_rate": float, "page_depth": float, "avg_duration_seconds": int}`

**DB functions (async, AsyncSession):**

3. `async def save_daily_snapshots(db: AsyncSession, site_id: UUID, rows: list[dict]) -> int`
   - Upserts rows into `metrika_traffic_daily` using `INSERT ... ON CONFLICT (site_id, traffic_date) DO UPDATE`
   - Uses `sqlalchemy.dialects.postgresql.insert` for upsert
   - Returns count of rows upserted

4. `async def save_page_snapshots(db: AsyncSession, site_id: UUID, period_start: date, period_end: date, rows: list[dict]) -> int`
   - Upserts rows into `metrika_traffic_pages` using `INSERT ... ON CONFLICT (site_id, period_start, period_end, page_url) DO UPDATE`
   - Returns count of rows upserted

5. `async def get_daily_traffic(db: AsyncSession, site_id: UUID, date_from: date | None = None, date_to: date | None = None) -> list[dict]`
   - SELECT from metrika_traffic_daily WHERE site_id = :site_id AND traffic_date BETWEEN date_from AND date_to
   - If date_from is None, default to 90 days ago
   - If date_to is None, default to today
   - ORDER BY traffic_date ASC
   - Returns list of dicts

6. `async def get_page_traffic(db: AsyncSession, site_id: UUID, period_start: date, period_end: date) -> list[dict]`
   - SELECT from metrika_traffic_pages WHERE site_id AND period_start AND period_end match
   - ORDER BY visits DESC
   - Returns list of dicts

**Pure computation functions:**

7. `def compute_period_delta(rows_a: list[dict], rows_b: list[dict]) -> list[dict]`
   - Join on `page_url`
   - For each URL compute: visits_a, visits_b, visits_delta, bounce_rate_a, bounce_rate_b, page_depth_a, page_depth_b, avg_duration_a, avg_duration_b, is_new (visits_a==0 and visits_b>0), is_lost (visits_a>0 and visits_b==0)
   - Sort by visits_b descending
   - Returns list of dicts

**Event CRUD functions:**

8. `async def get_events(db: AsyncSession, site_id: UUID) -> list[MetrikaEvent]`
   - SELECT from metrika_events WHERE site_id ORDER BY event_date ASC

9. `async def create_event(db: AsyncSession, site_id: UUID, event_date: date, label: str, color: str = "#6b7280") -> MetrikaEvent`
   - INSERT new MetrikaEvent, flush, return it

10. `async def delete_event(db: AsyncSession, event_id: UUID) -> bool`
    - DELETE from metrika_events WHERE id, return True if deleted

```python
"""Celery tasks for on-demand Metrika data fetching."""
from __future__ import annotations
import uuid
from datetime import date, timedelta

from loguru import logger
from app.celery_app import celery_app


@celery_app.task(
    name="app.tasks.metrika_tasks.fetch_metrika_data",
    bind=True,
    max_retries=3,
    soft_time_limit=120,
    time_limit=150,
)
def fetch_metrika_data(self, site_id: str, date1: str, date2: str) -> dict:
    """Fetch Metrika traffic data for a site and save to DB.

    Triggered by user button click (D-08). Fetches daily aggregate
    and per-page traffic for the given date range.
    """
    import asyncio
    from app.database import get_sync_db
    from app.models.site import Site
    from app.services.crypto_service import decrypt
    from sqlalchemy import select

    with get_sync_db() as db:
        site = db.execute(
            select(Site).where(Site.id == uuid.UUID(site_id))
        ).scalar_one_or_none()

    if not site:
        return {"status": "error", "reason": "site not found"}

    if not site.metrika_counter_id or not site.metrika_token:
        return {"status": "error", "reason": "metrika not configured"}

    counter_id = site.metrika_counter_id
    token = decrypt(site.metrika_token)

    try:
        loop = asyncio.new_event_loop()
        result = loop.run_until_complete(
            _fetch_and_save(site_id, counter_id, token, date1, date2)
        )
        loop.close()
        return result
    except Exception as exc:
        logger.error("Metrika fetch failed", site_id=site_id, error=str(exc))
        if "420" in str(exc) or "429" in str(exc):
            raise self.retry(countdown=60, exc=exc)
        raise self.retry(countdown=10, exc=exc)


async def _fetch_and_save(
    site_id: str, counter_id: str, token: str, date1: str, date2: str
) -> dict:
    from app.database import AsyncSessionLocal
    from app.services.metrika_service import (
        fetch_daily_traffic,
        fetch_page_traffic,
        save_daily_snapshots,
        save_page_snapshots,
    )
    from datetime import date as date_type

    daily_rows = await fetch_daily_traffic(counter_id, token, date1, date2)
    page_rows = await fetch_page_traffic(counter_id, token, date1, date2)

    sid = uuid.UUID(site_id)
    p_start = date_type.fromisoformat(date1)
    p_end = date_type.fromisoformat(date2)

    async with AsyncSessionLocal() as db:
        daily_count = await save_daily_snapshots(db, sid, daily_rows)
        page_count = await save_page_snapshots(db, sid, p_start, p_end, page_rows)
        await db.commit()

    logger.info(
        "Metrika data saved",
        site_id=site_id,
        daily_rows=daily_count,
        page_rows=page_count,
    )
    return {
        "status": "done",
        "site_id": site_id,
        "daily_rows": daily_count,
        "page_rows": page_count,
    }
```

Key design points:
- Uses sync Celery task with internal async event loop (same pattern as position_tasks.py)
- Decrypts metrika_token via crypto_service before passing to API functions
- Retries with 60s countdown on HTTP 420 (rate limit), 10s on other errors
- Max 3 retries per project constraint

```python
from app.services.metrika_service import compute_period_delta


def test_compute_period_delta_basic():
    rows_a = [
        {"page_url": "/page-1/", "visits": 100, "bounce_rate": 30.0, "page_depth": 3.0, "avg_duration_seconds": 120},
        {"page_url": "/page-2/", "visits": 50, "bounce_rate": 40.0, "page_depth": 2.0, "avg_duration_seconds": 90},
    ]
    rows_b = [
        {"page_url": "/page-1/", "visits": 150, "bounce_rate": 25.0, "page_depth": 3.5, "avg_duration_seconds": 140},
        {"page_url": "/page-3/", "visits": 30, "bounce_rate": 50.0, "page_depth": 1.5, "avg_duration_seconds": 60},
    ]
    result = compute_period_delta(rows_a, rows_b)
    assert len(result) == 3  # page-1, page-2, page-3

    by_url = {r["page_url"]: r for r in result}

    # page-1: growth
    assert by_url["/page-1/"]["visits_delta"] == 50
    assert by_url["/page-1/"]["is_new"] is False
    assert by_url["/page-1/"]["is_lost"] is False

    # page-2: lost
    assert by_url["/page-2/"]["visits_b"] == 0
    assert by_url["/page-2/"]["is_lost"] is True

    # page-3: new
    assert by_url["/page-3/"]["visits_a"] == 0
    assert by_url["/page-3/"]["is_new"] is True


def test_compute_period_delta_empty():
    result = compute_period_delta([], [])
    assert result == []


def test_compute_period_delta_sorted_by_visits_b():
    rows_a = [{"page_url": "/a/", "visits": 10, "bounce_rate": None, "page_depth": None, "avg_duration_seconds": None}]
    rows_b = [
        {"page_url": "/a/", "visits": 5, "bounce_rate": None, "page_depth": None, "avg_duration_seconds": None},
        {"page_url": "/b/", "visits": 20, "bounce_rate": None, "page_depth": None, "avg_duration_seconds": None},
    ]
    result = compute_period_delta(rows_a, rows_b)
    assert result[0]["page_url"] == "/b/"  # highest visits_b first
    assert result[1]["page_url"] == "/a/"
```
