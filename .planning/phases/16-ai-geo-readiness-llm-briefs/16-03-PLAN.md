    ```python
    from app.celery_app import celery_app
    from app.services.llm import llm_service
    from app.services.llm.config import ANTHROPIC_MODEL
    from app.services.user_service import get_anthropic_api_key
    from app.services.brief_service import get_brief_with_context  # use existing or nearest equivalent
    from app.models.llm_brief_job import LLMBriefJob
    import asyncio

    # CLAUDE.md mandates retry=3 for all external API calls.
    # Transient errors (network, timeout, 429, 529 overloaded) -> self.retry with exponential backoff.
    # Permanent errors (401/403 auth, 400 bad request, quota exceeded) -> fail fast, no retry.
    # Circuit breaker (record_llm_failure) only fires on REAL failure (after retries exhausted OR permanent error),
    # not on every transient retry ŌĆö otherwise the breaker would open from a single flaky network blip.
    @celery_app.task(name="llm.generate_brief_enhancement", bind=True, max_retries=3)
    def generate_llm_brief_enhancement(self, job_id: int) -> None:
        try:
            asyncio.run(_run(self, job_id))
        except _TransientLLMError as exc:
            # countdown = 60 * 2 ** retries -> 60s, 120s, 240s
            raise self.retry(exc=exc, countdown=60 * 2 ** self.request.retries)

    class _TransientLLMError(Exception):
        """Raised to signal Celery should retry via self.retry ŌĆö does NOT trip circuit breaker."""
        pass

    async def _run(task, job_id: int) -> None:
        async with AsyncSessionLocal() as session:  # use existing session factory
            job = await session.get(LLMBriefJob, job_id)
            if not job:
                return
            job.status = "running"
            await session.commit()

            user = await session.get(User, job.user_id)
            redis = get_redis()  # existing redis getter

            # 1. Circuit check
            if await llm_service.is_circuit_open(redis, user.id):
                await _fail(session, job, user, "LLM circuit open ŌĆö try again in 15 minutes", redis=redis)
                return

            # 2. API key
            api_key = await get_anthropic_api_key(session, user)
            if not api_key:
                await _fail(session, job, user, "Anthropic API key not configured", redis=redis)
                return

            # 3. Build context + prompt
            brief, context = await get_brief_with_context(session, job.brief_id)
            prompt = llm_service.build_brief_prompt(brief, context)

            # 4. Call API ŌĆö distinguish transient vs permanent errors
            # Import inside the function so we can reference Anthropic exception classes
            from anthropic import APIConnectionError, APITimeoutError, RateLimitError, APIStatusError, AuthenticationError, PermissionDeniedError, BadRequestError
            try:
                output, in_tok, out_tok = await llm_service.call_llm_brief_enhance(api_key, prompt)
            except (AuthenticationError, PermissionDeniedError, BadRequestError) as exc:
                # PERMANENT ŌĆö fail fast, count toward circuit breaker, do NOT retry
                await llm_service.record_llm_failure(redis, user.id)
                await _fail(session, job, user, f"Anthropic permanent error: {exc}", redis=redis, in_tok=0, out_tok=0)
                return
            except (APIConnectionError, APITimeoutError, RateLimitError) as exc:
                # TRANSIENT ŌĆö surface to outer task wrapper for self.retry; do NOT touch circuit breaker yet
                # Only on the FINAL retry exhaustion does this become a real failure (handled below).
                if task.request.retries >= task.max_retries:
                    await llm_service.record_llm_failure(redis, user.id)
                    await _fail(session, job, user, f"Anthropic transient error after {task.max_retries} retries: {exc}", redis=redis, in_tok=0, out_tok=0)
                    return
                raise _TransientLLMError(str(exc)) from exc
            except APIStatusError as exc:
                # 5xx (excluding 529 which is RateLimitError-classed by SDK) ŌĆö treat as transient
                if 500 <= exc.status_code < 600 and task.request.retries < task.max_retries:
                    raise _TransientLLMError(str(exc)) from exc
                await llm_service.record_llm_failure(redis, user.id)
                await _fail(session, job, user, f"Anthropic API error {exc.status_code}: {exc}", redis=redis, in_tok=0, out_tok=0)
                return
            except Exception as exc:
                # Unknown ŌĆö treat as permanent to be safe
                await llm_service.record_llm_failure(redis, user.id)
                await _fail(session, job, user, f"Anthropic error: {exc}", redis=redis, in_tok=0, out_tok=0)
                return

            # 5. Success
            job.status = "done"
            job.output_json = output
            await session.commit()
            await llm_service.record_llm_success(redis, user.id)
            await llm_service.log_llm_usage(
                session, user_id=user.id, brief_id=job.brief_id, job_id=job.id,
                model=ANTHROPIC_MODEL, input_tokens=in_tok, output_tokens=out_tok,
                status="success",
            )

    async def _fail(session, job, user, message, *, redis, in_tok=0, out_tok=0):
        job.status = "failed"
        job.error_message = message
        await session.commit()
        await llm_service.log_llm_usage(
            session, user_id=user.id, brief_id=job.brief_id, job_id=job.id,
            model=ANTHROPIC_MODEL, input_tokens=in_tok, output_tokens=out_tok,
            status="failed", error_message=message,
        )
    ```
