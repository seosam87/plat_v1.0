    ```python
    def fetch_yandex_suggest_sync(query: str, proxy_url: str | None = None, timeout: int = 10) -> list[str]:
        """Fetch Yandex Suggest results for a single query. Returns list of suggestion strings."""
        params = {"part": query, "uil": "ru", "lr": "213", "highlight": "0", "v": "4", "sn": "5", "srv": "suggest_ya_search"}
        proxies = {"http://": proxy_url, "https://": proxy_url} if proxy_url else None
        with httpx.Client(timeout=timeout, proxy=proxy_url) as client:
            resp = client.get(YANDEX_SUGGEST_URL, params=params)
            resp.raise_for_status()
            data = resp.json()
            # Response is typically: ["query", ["suggestion1", "suggestion2", ...]]
            if isinstance(data, list) and len(data) > 1 and isinstance(data[1], list):
                return [str(s) for s in data[1]]
            # Alternative format: {"items": [{"value": "..."}, ...]}
            if isinstance(data, dict) and "items" in data:
                return [item["value"] for item in data["items"]]
            return []
        # On any exception: log warning, return []
    ```

    ```python
    def fetch_google_suggest_sync(query: str, timeout: int = 10) -> list[str]:
        """Fetch Google Suggest results. No proxy needed (D-07)."""
        params = {"client": "chrome", "q": query, "hl": "ru", "gl": "ru"}
        with httpx.Client(timeout=timeout) as client:
            resp = client.get(GOOGLE_SUGGEST_URL, params=params)
            resp.raise_for_status()
            data = resp.json()
            # Response: ["query", ["suggestion1", "suggestion2", ...], ...]
            return [str(s) for s in data[1]] if len(data) > 1 else []
        # On exception: log warning, return []
    ```

    ```python
    def suggest_cache_key(seed: str, include_google: bool) -> str:
        normalized = seed.strip().lower()
        sources = "yg" if include_google else "y"
        return f"suggest:{sources}:{normalized}"
    ```

    ```python
    def deduplicate_suggestions(
        yandex_results: list[str], google_results: list[str]
    ) -> list[dict]:
        """Merge and deduplicate. Returns list of {"keyword": str, "source": "yandex"|"google"}."""
        seen = set()
        result = []
        for kw in yandex_results:
            norm = kw.strip().lower()
            if norm and norm not in seen:
                seen.add(norm)
                result.append({"keyword": kw.strip(), "source": "yandex"})
        for kw in google_results:
            norm = kw.strip().lower()
            if norm and norm not in seen:
                seen.add(norm)
                result.append({"keyword": kw.strip(), "source": "google"})
        return result
    ```

    ```python
    def get_active_proxy_urls_sync() -> list[str]:
        """Return URLs of all active proxies from DB (for Yandex Suggest rotation)."""
        from app.database import get_sync_db
        from app.models.proxy import Proxy, ProxyStatus
        from sqlalchemy import select
        with get_sync_db() as db:
            proxies = db.execute(
                select(Proxy.url).where(Proxy.status == ProxyStatus.active, Proxy.is_active == True)
            ).scalars().all()
        return list(proxies)
    ```

    **Celery task** (`app/tasks/suggest_tasks.py`):

    ```python
    import json
    import random
    import time
    import uuid

    import redis as sync_redis
    from loguru import logger

    from app.celery_app import celery_app
    from app.config import settings
    from app.database import get_sync_db
    from app.models.suggest_job import SuggestJob
    from app.services.suggest_service import (
        RU_ALPHABET,
        SUGGEST_CACHE_TTL,
        deduplicate_suggestions,
        fetch_google_suggest_sync,
        fetch_yandex_suggest_sync,
        get_active_proxy_urls_sync,
        suggest_cache_key,
    )

    @celery_app.task(
        name="app.tasks.suggest_tasks.fetch_suggest_keywords",
        bind=True,
        max_retries=3,
        queue="default",
        soft_time_limit=300,
        time_limit=360,
    )
    def fetch_suggest_keywords(self, job_id: str) -> dict:
        """Full alphabetic expansion suggest fetch. Checks Redis cache first."""
        r = sync_redis.from_url(settings.REDIS_URL, decode_responses=True)

        # Load job from DB
        with get_sync_db() as db:
            job = db.get(SuggestJob, uuid.UUID(job_id))
            if not job:
                return {"status": "failed", "error": "Job not found"}
            seed = job.seed
            include_google = job.include_google
            cache_key_val = job.cache_key or suggest_cache_key(seed, include_google)
            job.status = "running"
            job.cache_key = cache_key_val
            db.commit()

        # Check Redis cache
        cached = r.get(cache_key_val)
        if cached:
            data = json.loads(cached)
            with get_sync_db() as db:
                job = db.get(SuggestJob, uuid.UUID(job_id))
                job.status = "complete"
                job.cache_hit = True
                job.result_count = len(data)
                db.commit()
            return {"status": "complete", "cache_hit": True, "count": len(data)}

        # Alphabetic expansion — Yandex (with proxy rotation per D-05, D-06, D-17)
        proxies = get_active_proxy_urls_sync()
        yandex_all = []
        was_banned = False
        proxy_idx = 0

        for letter in RU_ALPHABET:
            query = f"{seed} {letter}"
            pause = random.uniform(0.2, 0.5)  # D-17: 200-500ms pause
            time.sleep(pause)

            fetched = False
            for attempt in range(3):  # D-06: max 3 attempts per letter
                proxy_url = proxies[proxy_idx % len(proxies)] if proxies else None
                results = fetch_yandex_suggest_sync(query, proxy_url=proxy_url)
                if results:
                    yandex_all.extend(results)
                    fetched = True
                    break
                # Rotate proxy on failure
                proxy_idx += 1
                if proxy_idx >= len(proxies) * 3:
                    was_banned = True
                    break
                time.sleep(30)  # D-06: 30s pause before retry

            if was_banned:
                logger.warning("All proxies exhausted at letter '{}' for seed '{}'", letter, seed)
                break

        # Google expansion (no proxy, per D-07)
        google_all = []
        if include_google:
            for letter in RU_ALPHABET:
                query = f"{seed} {letter}"
                time.sleep(random.uniform(0.2, 0.5))
                results = fetch_google_suggest_sync(query)
                google_all.extend(results)

        # Deduplicate and cache
        combined = deduplicate_suggestions(yandex_all, google_all)

        if combined:
            r.set(cache_key_val, json.dumps(combined, ensure_ascii=False), ex=SUGGEST_CACHE_TTL)

        # Update job status
        status = "partial" if was_banned and combined else ("failed" if not combined else "complete")
        with get_sync_db() as db:
            job = db.get(SuggestJob, uuid.UUID(job_id))
            job.status = status
            job.result_count = len(combined)
            job.expected_count = 33 * 10  # ~330 expected from full expansion
            if was_banned:
                job.error_message = "Прокси заблокированы, получены частичные результаты"
            if not combined:
                job.error_message = "Не удалось получить подсказки — все прокси заблокированы"
            db.commit()

        return {"status": status, "count": len(combined), "was_banned": was_banned}
    ```

    **Important implementation notes:**
    - Use `redis.from_url` (sync), NOT `redis.asyncio` — Celery tasks are sync per anti-pattern in RESEARCH.md
    - If no proxies in DB, fall back to direct request (proxy_url=None) — may get banned quickly but still functional
    - The proxy rotation logic: cycle through available proxies, with `proxy_idx` tracking position; after 3x full rotations through the pool, consider proxies exhausted
