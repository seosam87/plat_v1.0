```python
"""QA Surface Tracker — admin UI for managing feature coverage across surfaces.

Routes are under /ui/qa/ prefix. All endpoints require authentication.
Per D-05: discover_routes() called at request time ONLY (never at import time)
to avoid circular import with app.main.
"""
from __future__ import annotations
import uuid
from datetime import datetime, timezone

from fastapi import APIRouter, Depends, Form, Request
from fastapi.responses import HTMLResponse, RedirectResponse
from sqlalchemy.ext.asyncio import AsyncSession

from app.dependencies import get_db, get_current_user
from app.models.qa_surface import Surface, CheckStatus
from app.services import qa_surface_service as svc
from app.template_engine import templates

router = APIRouter(prefix="/ui/qa", tags=["qa"])

STATUS_COLORS = {
    "not_tested":   {"bg": "#f3f4f6", "text": "#6b7280"},
    "passed":       {"bg": "#d1fae5", "text": "#065f46"},
    "failed":       {"bg": "#fee2e2", "text": "#991b1b"},
    "needs_retest": {"bg": "#fef3c7", "text": "#92400e"},
}
```

Endpoints to implement:

1. `GET /ui/qa/` — Matrix view. Calls `svc.list_features_with_checks(db)`. Renders `qa/index.html` with `features`, `surfaces=list(Surface)`, `status_colors=STATUS_COLORS`, `CheckStatus=CheckStatus`.

2. `GET /ui/qa/features/new` — Create form. Renders `qa/feature_form.html` with `feature=None`, `mode="create"`.

3. `POST /ui/qa/features` — Create feature. Reads Form fields: `name: str`, `slug: str`, `description: str = ""`, `retest_days: int = 30`. Calls `svc.create_feature_surface(db, slug, name, description or None, retest_days)`. Redirects to `/ui/qa/`.

4. `GET /ui/qa/features/{feature_surface_id}` — Edit form. Calls `svc.get_feature_by_id(db, feature_surface_id)`. Renders `qa/feature_form.html` with `feature=feature`, `mode="edit"`.

5. `POST /ui/qa/features/{feature_surface_id}` — Update feature. Reads Form fields same as create. Calls `svc.update_feature_surface(db, feature_surface_id, name, slug, description or None, retest_days)`. Redirects to `/ui/qa/`.

6. `POST /ui/qa/features/{feature_surface_id}/delete` — Delete feature. Calls `svc.delete_feature_surface(db, feature_surface_id)`. Redirects to `/ui/qa/`.

7. `GET /ui/qa/checks/{surface_check_id}/edit` — HTMX partial for inline check editing. Calls `svc.get_check_by_id(db, surface_check_id)`. Renders `qa/_check_cell.html` with `check=check`, `statuses=list(CheckStatus)`, `mode="edit"`, `status_colors=STATUS_COLORS`.

8. `POST /ui/qa/checks/{surface_check_id}/mark-tested` — Mark check. Reads Form: `status: str`, `notes: str = ""`. Gets username from `user.username`. Calls `svc.mark_check_tested(db, surface_check_id, CheckStatus(status), notes or None, user.username)`. Renders `qa/_check_cell.html` with `check=updated_check`, `mode="view"`, `status_colors=STATUS_COLORS`.

9. `GET /ui/qa/checks/{surface_check_id}` — HTMX partial for viewing (restores cell after edit cancel). Calls `svc.get_check_by_id(db, surface_check_id)`. Renders `qa/_check_cell.html` with `check=check`, `mode="view"`, `status_colors=STATUS_COLORS`.

```python
async def candidates_page(request: Request, db: AsyncSession = Depends(get_db), user=Depends(get_current_user)):
```
