```javascript
function initTwoTapButton(btn) {
  let timer = null;
  btn.addEventListener('click', function(e) {
    e.preventDefault();
    if (btn.dataset.ready === 'true') {
      htmx.ajax('POST', btn.dataset.actionUrl, {target: btn.dataset.target, swap: 'outerHTML'});
      btn.dataset.ready = 'false';
      return;
    }
    const originalText = btn.textContent;
    const originalClass = btn.className;
    btn.textContent = btn.dataset.confirmText;
    btn.className = btn.className.replace(/border-gray-300 text-gray-700 bg-white/, btn.dataset.confirmClass);
    btn.dataset.ready = 'true';
    timer = setTimeout(function() {
      btn.textContent = originalText;
      btn.className = originalClass;
      btn.dataset.ready = 'false';
    }, 2000);
  });
}
document.addEventListener('htmx:afterSettle', function() {
  document.querySelectorAll('[data-confirm-text]:not([data-initialized])').forEach(function(btn) {
    initTwoTapButton(btn);
    btn.dataset.initialized = 'true';
  });
});
document.querySelectorAll('[data-confirm-text]').forEach(function(btn) {
  initTwoTapButton(btn);
  btn.dataset.initialized = 'true';
});
```
