```python
"""Tests for subprocess-isolated WeasyPrint PDF renderer."""
import os
import tempfile
from unittest.mock import patch

import pytest

from app.services.subprocess_pdf import render_pdf_in_subprocess


class TestRenderPdfInSubprocess:
    """Test subprocess PDF rendering."""

    def test_valid_html_returns_pdf_bytes(self):
        """Minimal valid HTML produces PDF bytes starting with %PDF."""
        html = "<html><body><h1>Test</h1></body></html>"
        result = render_pdf_in_subprocess(html)
        assert isinstance(result, bytes)
        assert result[:5] == b"%PDF-"
        assert len(result) > 100  # non-trivial PDF

    def test_russian_text_renders(self):
        """Russian text in HTML renders without error."""
        html = "<html><body><h1>Тестовый отчёт</h1><p>Проверка кириллицы</p></body></html>"
        result = render_pdf_in_subprocess(html)
        assert result[:5] == b"%PDF-"

    def test_timeout_raises_runtime_error(self):
        """Very short timeout with complex HTML raises RuntimeError."""
        # Use a huge HTML to exceed timeout
        html = "<html><body>" + "<p>x</p>" * 100000 + "</body></html>"
        with pytest.raises(RuntimeError, match="timed out|failed"):
            render_pdf_in_subprocess(html, timeout_seconds=1)

    def test_temp_files_cleaned_on_success(self):
        """Temp HTML and PDF files are removed after successful render."""
        html = "<html><body><p>cleanup test</p></body></html>"
        # Before: count temp files
        temp_dir = tempfile.gettempdir()
        before = set(os.listdir(temp_dir))
        render_pdf_in_subprocess(html)
        after = set(os.listdir(temp_dir))
        # No new .html or .pdf files should remain
        new_files = after - before
        leftover = [f for f in new_files if f.endswith(('.html', '.pdf'))]
        assert len(leftover) == 0, f"Leftover temp files: {leftover}"

    def test_subprocess_failure_raises_runtime_error(self):
        """If the subprocess script fails, RuntimeError is raised."""
        with patch("app.services.subprocess_pdf.subprocess.run") as mock_run:
            mock_run.return_value.returncode = 1
            mock_run.return_value.stderr = "ImportError: no module named weasyprint"
            with pytest.raises(RuntimeError, match="PDF render failed"):
                render_pdf_in_subprocess("<html></html>")
```
