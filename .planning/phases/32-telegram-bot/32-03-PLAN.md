All handlers decorated with `@require_auth` from bot.auth.

`status_handler` — /status command:
- Runs `docker compose ps --format json` via `run_command(["docker", "compose", "ps", "--format", "json"])` with timeout=15
- Parses JSON output, builds status_line() for each service (name, state, health)
- Falls back to simple text if docker not available
- Adds contextual WebApp button: make_webapp_button("Открыть дайджест", "/m/digest", settings.APP_BASE_URL)
- Reply with parse_mode="HTML"

`logs_handler` — /logs command:
- Runs `docker compose logs --tail=50 api worker` via run_command with timeout=15
- Returns last 50 lines wrapped in code_block()
- No confirmation needed (read-only)

`test_handler` — /test command (dangerous per D-07):
- Shows confirmation inline keyboard: InlineKeyboardMarkup with "Выполнить" (callback_data="confirm:test") and "Отмена" (callback_data="cancel:test")
- On confirm: dispatches via Celery `dispatch("app.tasks.devops_tasks.run_tests", queue="default")` OR runs `run_command(["python", "-m", "pytest", "--tb=short", "-q"], timeout=120)`
- Decision: use run_command directly since pytest runs in the same codebase
- Auto-cancel timeout 60s via asyncio.create_task

`deploy_handler` — /deploy command (dangerous per D-07):
- Shows confirmation inline keyboard: "Выполнить" (callback_data="confirm:deploy") / "Отмена" (callback_data="cancel:deploy")
- On confirm: runs `run_command(["git", "pull"], timeout=30)` then sends result
- Auto-cancel timeout 60s

Module-level dict `_pending: dict[int, asyncio.Task] = {}` for tracking auto-cancel tasks by message_id.

`confirm_callback` — CallbackQueryHandler for "confirm:*":
- `await query.answer("Выполняю...")`
- Cancel pending timeout task from _pending dict
- Parse operation from callback_data.split(":")[1]
- Execute the operation (test or deploy) and edit message with result

`cancel_callback` — CallbackQueryHandler for "cancel:*":
- `await query.answer()`
- Cancel pending timeout task
- Edit message: "Операция отменена."

**5. bot/main.py** — Entry point per D-02:

```python
import sys
import os
# Ensure project root is in sys.path so `from app.models...` works
sys.path.insert(0, os.path.dirname(os.path.dirname(os.path.abspath(__file__))))

from loguru import logger
from telegram import MenuButtonWebApp, WebAppInfo
from telegram.ext import Application, CommandHandler, CallbackQueryHandler

from bot.config import BotSettings
from bot.handlers.devops import (
    status_handler, logs_handler, test_handler, deploy_handler,
    confirm_callback, cancel_callback,
)
from bot.handlers.seo import (
    crawl_handler, check_handler, report_handler, seo_site_callback,
)
from bot.handlers.miniapp import start_handler, help_handler

settings = BotSettings()

# Configure loguru
logger.remove()
logger.add(
    sys.stderr,
    level=settings.LOG_LEVEL,
    format="{time:YYYY-MM-DD HH:mm:ss} | {level} | {message}",
)

async def post_init(application: Application) -> None:
    """One-time setup: Menu Button + command list."""
    await application.bot.set_chat_menu_button(
        menu_button=MenuButtonWebApp(
            text="Открыть платформу",
            web_app=WebAppInfo(url=f"{settings.APP_BASE_URL}/m/"),
        )
    )
    await application.bot.set_my_commands([
        ("status", "Статус сервисов"),
        ("logs", "Последние логи"),
        ("test", "Запустить тесты"),
        ("deploy", "Деплой (git pull + restart)"),
        ("crawl", "Запустить краул сайта"),
        ("check", "Проверить позиции"),
        ("report", "Сформировать отчёт"),
        ("help", "Помощь"),
    ])
    logger.info("Bot initialized, webhook commands registered")

async def error_handler(update, context):
    logger.error(f"Update {update} caused error: {context.error}")

def main():
    app = (
        Application.builder()
        .token(settings.TELEGRAM_BOT_TOKEN)
        .post_init(post_init)
        .build()
    )

    # Command handlers
    app.add_handler(CommandHandler("start", start_handler))
    app.add_handler(CommandHandler("help", help_handler))
    app.add_handler(CommandHandler("status", status_handler))
    app.add_handler(CommandHandler("logs", logs_handler))
    app.add_handler(CommandHandler("test", test_handler))
    app.add_handler(CommandHandler("deploy", deploy_handler))
    app.add_handler(CommandHandler("crawl", crawl_handler))
    app.add_handler(CommandHandler("check", check_handler))
    app.add_handler(CommandHandler("report", report_handler))

    # Callback handlers (confirmation buttons, site selection)
    app.add_handler(CallbackQueryHandler(confirm_callback, pattern=r"^confirm:"))
    app.add_handler(CallbackQueryHandler(cancel_callback, pattern=r"^cancel:"))
    app.add_handler(CallbackQueryHandler(seo_site_callback, pattern=r"^(crawl|check|report):\d+$"))

    app.add_error_handler(error_handler)

    logger.info(f"Starting bot webhook on port {settings.TELEGRAM_BOT_PORT}")
    app.run_webhook(
        listen="0.0.0.0",
        port=settings.TELEGRAM_BOT_PORT,
        url_path="/webhook/tg",
        secret_token=settings.TELEGRAM_WEBHOOK_SECRET,
        webhook_url=f"{settings.TELEGRAM_WEBHOOK_BASE_URL}/webhook/tg",
        drop_pending_updates=True,
    )

if __name__ == "__main__":
    main()
```

   Add new task:
   ```python
   @celery_app.task(
       name="app.tasks.notification_tasks.dispatch_tg_error_notification",
       bind=True,
       max_retries=3,
       default_retry_delay=10,
   )
   def dispatch_tg_error_notification(self, user_id: str, title: str, body: str) -> dict:
       """Send error notification to Telegram if user has telegram_id + tg_notifications_enabled=True."""
       from app.database import get_sync_session
       from app.config import settings
       
       with get_sync_session() as db:
           user = db.execute(
               select(User).where(User.id == user_id)
           ).scalar_one_or_none()
           
           if not user or not user.telegram_id or not user.tg_notifications_enabled:
               return {"sent": False, "reason": "user not eligible"}
           
           text = f"&#128308; <b>{title}</b>\n{body}"
           url = f"https://api.telegram.org/bot{settings.TELEGRAM_BOT_TOKEN}/sendMessage"
           
           try:
               resp = httpx.post(url, json={
                   "chat_id": user.telegram_id,
                   "text": text,
                   "parse_mode": "HTML",
               }, timeout=10)
               resp.raise_for_status()
               return {"sent": True}
           except Exception as exc:
               raise self.retry(exc=exc)
   ```

   ```python
   @router.post("/tg-notifications-toggle")
   async def toggle_tg_notifications(
       request: Request,
       db: AsyncSession = Depends(get_db),
       current_user: User = Depends(get_current_user),
   ):
       """Toggle Telegram notifications for current user (HTMX)."""
       current_user.tg_notifications_enabled = not current_user.tg_notifications_enabled
       await db.commit()
       
       status = "включены" if current_user.tg_notifications_enabled else "выключены"
       
       # Return HTMX partial or redirect
       if request.headers.get("HX-Request"):
           checked = "checked" if current_user.tg_notifications_enabled else ""
           return HTMLResponse(
               f'<label class="toggle-label">'
               f'<input type="checkbox" {checked} '
               f'hx-post="/profile/tg-notifications-toggle" '
               f'hx-swap="outerHTML" hx-target="closest label">'
               f' Уведомления в Telegram ({status})</label>'
           )
       return RedirectResponse("/profile", status_code=303)
   ```
